 91    def sanity_check(self) -> None:
 92        r"""Sanity check for config."""
 93
 94        # check required config fields
 95        required_config_fields = [
 96            "pipeline",
 97            "expr_name",
 98            "cl_paradigm",
 99            "train_tasks",
100            "eval_after_tasks",
101            "global_seed",
102            "cl_dataset",
103            "cl_algorithm",
104            "backbone",
105            "optimizer",
106            "trainer",
107            "metrics",
108            "lightning_loggers",
109            "callbacks",
110            "output_dir",
111            # "hydra" is excluded as it doesn't appear
112            "misc",
113        ]
114        for field in required_config_fields:
115            if not self.cfg.get(field):
116                raise KeyError(
117                    f"Field `{field}` is required in the experiment index config."
118                )
119
120        # check cl_paradigm
121        if self.cfg.cl_paradigm not in ["TIL", "CIL"]:
122            raise ValueError(
123                f"Field `cl_paradigm` should be either 'TIL' or 'CIL' but got {self.cfg.cl_paradigm}!"
124            )
125
126        # get dataset number of tasks
127        if self.cfg.cl_dataset.get("num_tasks"):
128            num_tasks = self.cfg.cl_dataset.get("num_tasks")
129        elif self.cfg.cl_dataset.get("class_split"):
130            num_tasks = len(self.cfg.cl_dataset.class_split)
131        elif self.cfg.cl_dataset.get("datasets"):
132            num_tasks = len(self.cfg.cl_dataset.datasets)
133        else:
134            raise KeyError(
135                "`num_tasks` is required in cl_dataset config. Please specify `num_tasks` (for `CLPermutedDataset`) or `class_split` (for `CLSplitDataset`) or `datasets` (for `CLCombinedDataset`) in cl_dataset config."
136            )
137
138        # check train_tasks
139        train_tasks = self.cfg.train_tasks
140        if isinstance(train_tasks, ListConfig):
141            if len(train_tasks) < 1:
142                raise ValueError("`train_tasks` config must contain at least one task.")
143            if any(t < 1 or t > num_tasks for t in train_tasks):
144                raise ValueError(
145                    f"All task IDs in `train_tasks` config must be between 1 and {num_tasks}."
146                )
147        elif isinstance(train_tasks, int):
148            if train_tasks < 0 or train_tasks > num_tasks:
149                raise ValueError(
150                    f"`train_tasks` config as integer must be between 0 and {num_tasks}."
151                )
152        else:
153            raise TypeError(
154                "`train_tasks` config must be either a list of integers or an integer."
155            )
156
157        # check eval_after_tasks
158        eval_after_tasks = self.cfg.eval_after_tasks
159        if isinstance(eval_after_tasks, ListConfig):
160            if len(eval_after_tasks) < 1:
161                raise ValueError(
162                    "`eval_after_tasks` config must contain at least one task."
163                )
164            if any(t < 1 or t > num_tasks for t in eval_after_tasks):
165                raise ValueError(
166                    f"All task IDs in `eval_after_tasks` config must be between 1 and {num_tasks}."
167                )
168        elif isinstance(eval_after_tasks, int):
169            if eval_after_tasks < 0 or eval_after_tasks > num_tasks:
170                raise ValueError(
171                    f"`eval_after_tasks` config as integer must be between 0 and {num_tasks}."
172                )
173        else:
174            raise TypeError(
175                "`eval_after_tasks` config must be either a list of integers or an integer."
176            )
177
178        # check that eval_after_tasks is a subset of train_tasks
179        if isinstance(train_tasks, list) and isinstance(eval_after_tasks, list):
180            if not set(eval_after_tasks).issubset(set(train_tasks)):
181                raise ValueError(
182                    "`eval_after_tasks` config must be a subset of `train_tasks` config."
183                )

376    def run(self) -> None:
377        r"""The main method to run the continual learning main experiment."""
378
379        self.set_global_seed(self.global_seed)
380
381        # global components
382        self.instantiate_cl_dataset(cl_dataset_cfg=self.cfg.cl_dataset)
383        self.cl_dataset.set_cl_paradigm(cl_paradigm=self.cl_paradigm)
384        self.instantiate_backbone(backbone_cfg=self.cfg.backbone)
385        self.instantiate_heads(
386            cl_paradigm=self.cl_paradigm, input_dim=self.cfg.backbone.output_dim
387        )
388        self.instantiate_cl_algorithm(
389            cl_algorithm_cfg=self.cfg.cl_algorithm,
390            backbone=self.backbone,
391            heads=self.heads,
392            non_algorithmic_hparams=select_hyperparameters_from_config(
393                cfg=self.cfg, type=self.cfg.pipeline
394            ),
395        )  # cl_algorithm should be instantiated after backbone and heads
396        self.instantiate_lightning_loggers(
397            lightning_loggers_cfg=self.cfg.lightning_loggers
398        )
399        self.instantiate_callbacks(
400            metrics_cfg=self.cfg.metrics,
401            callbacks_cfg=self.cfg.callbacks,
402        )
403
404        # task loop
405        for task_id in self.train_tasks:
406
407            self.task_id = task_id
408
409            # task-specific components
410            self.instantiate_optimizer(
411                optimizer_cfg=self.cfg.optimizer,
412                task_id=task_id,
413            )
414            if self.cfg.get("lr_scheduler"):
415                self.instantiate_lr_scheduler(
416                    lr_scheduler_cfg=self.cfg.lr_scheduler,
417                    task_id=task_id,
418                )
419            self.instantiate_trainer(
420                trainer_cfg=self.cfg.trainer,
421                lightning_loggers=self.lightning_loggers,
422                callbacks=self.callbacks,
423                task_id=task_id,
424            )  # trainer should be instantiated after lightning loggers and callbacks
425
426            # setup task ID for dataset and model
427            self.cl_dataset.setup_task_id(task_id=task_id)
428            self.model.setup_task_id(
429                task_id=task_id,
430                num_classes=len(self.cl_dataset.get_cl_class_map(self.task_id)),
431                optimizer=self.optimizer_t,
432                lr_scheduler=self.lr_scheduler_t,
433            )
434
435            # train and validate the model
436            self.trainer_t.fit(
437                model=self.model,
438                datamodule=self.cl_dataset,
439            )
440
441            # evaluation after training and validation
442            if task_id in self.eval_after_tasks:
443                self.trainer_t.test(
444                    model=self.model,
445                    datamodule=self.cl_dataset,
446                )
447
448            self.processed_task_ids.append(task_id)

377    def run(self) -> None:
378        r"""The main method to run the continual learning main experiment."""
379
380        self.set_global_seed(self.global_seed)
381
382        # global components
383        self.instantiate_cl_dataset(cl_dataset_cfg=self.cfg.cl_dataset)
384        self.cl_dataset.set_cl_paradigm(cl_paradigm=self.cl_paradigm)
385        self.instantiate_backbone(backbone_cfg=self.cfg.backbone)
386        self.instantiate_heads(
387            cl_paradigm=self.cl_paradigm, input_dim=self.cfg.backbone.output_dim
388        )
389        self.instantiate_cl_algorithm(
390            cl_algorithm_cfg=self.cfg.cl_algorithm,
391            backbone=self.backbone,
392            heads=self.heads,
393            non_algorithmic_hparams=select_hyperparameters_from_config(
394                cfg=self.cfg, type=self.cfg.pipeline
395            ),
396        )  # cl_algorithm should be instantiated after backbone and heads
397        self.instantiate_lightning_loggers(
398            lightning_loggers_cfg=self.cfg.lightning_loggers
399        )
400        self.instantiate_callbacks(
401            metrics_cfg=self.cfg.metrics,
402            callbacks_cfg=self.cfg.callbacks,
403        )
404
405        # task loop
406        for task_id in self.train_tasks:
407
408            self.task_id = task_id
409
410            # task-specific components
411            self.instantiate_optimizer(
412                optimizer_cfg=self.cfg.optimizer,
413                task_id=task_id,
414            )
415            if self.cfg.get("lr_scheduler"):
416                self.instantiate_lr_scheduler(
417                    lr_scheduler_cfg=self.cfg.lr_scheduler,
418                    task_id=task_id,
419                )
420            self.instantiate_trainer(
421                trainer_cfg=self.cfg.trainer,
422                lightning_loggers=self.lightning_loggers,
423                callbacks=self.callbacks,
424                task_id=task_id,
425            )  # trainer should be instantiated after lightning loggers and callbacks
426
427            # setup task ID for dataset and model
428            self.cl_dataset.setup_task_id(task_id=task_id)
429            self.model.setup_task_id(
430                task_id=task_id,
431                num_classes=len(self.cl_dataset.get_cl_class_map(self.task_id)),
432                optimizer=self.optimizer_t,
433                lr_scheduler=self.lr_scheduler_t,
434            )
435
436            # train and validate the model
437            self.trainer_t.fit(
438                model=self.model,
439                datamodule=self.cl_dataset,
440            )
441
442            # evaluation after training and validation
443            if task_id in self.eval_after_tasks:
444                self.trainer_t.test(
445                    model=self.model,
446                    datamodule=self.cl_dataset,
447                )
448
449            self.processed_task_ids.append(task_id)
